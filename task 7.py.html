class Book:
    """A class representing a book."""

    def __init__(self, title, author, publisher, page_count, price):
        self.title = title
        self.author = author
        self.publisher = publisher
        self.page_count = page_count
        self.price = price

    def __str__(self):
        return (f"Title: {self.title}, Author: {self.author}, Publisher: {self.publisher}, "
                f"Pages: {self.page_count}, Price: Â£{self.price:.2f}")

def main_book():
    # List of books from the provided data
    book_list = [
        ["The Great Gatsby", "F. Scott Fitzgerald", "Charles Scribner's Sons", 218, 10.99],
        ["1984", "George Orwell", "Secker & Warburg", 328, 8.99],
        ["To Kill a Mockingbird", "Harper Lee", "J.B. Lippincott & Co.", 281, 7.99],
        ["Pride and Prejudice", "Jane Austen", "T. Egerton", 279, 9.50],
        ["The Catcher in the Rye", "J.D. Salinger", "Little, Brown and Company", 214, 6.99]
    ]

    # Creating a list of Book objects
    books = [Book(*book) for book in book_list]

    # Display book details
    print("Books in the collection:")
    for book in books:
        print(book)

    # Add a new book to the text file and collection
    new_book = Book("Moby Dick", "Herman Melville", "Harper & Brothers", 635, 11.50)
    books.append(new_book)
    print("\nAfter adding a new book:")
    for book in books:
        print(book)


if __name__ == "__main__":
    main_book()

class EBook(Book):
    """A subclass representing an eBook."""

    def __init__(self, title, author, publisher, page_count, price, file_size, file_format):
        super().__init__(title, author, publisher, page_count, price)
        self.file_size = file_size
        self.file_format = file_format

    def __str__(self):
        return (super().__str__() +
                f", File Size: {self.file_size}MB, Format: {self.file_format}")


# Test program for the EBook class
def main_ebook():
    ebooks = []
    print("Enter details of eBooks (type 'stop' to finish):")
    while True:
        title = input("Title: ").strip()
        if title.lower() == "stop":
            break
        author = input("Author: ").strip()
        publisher = input("Publisher: ").strip()
        try:
            page_count = int(input("Page count: ").strip())
            price = float(input("Price: ").strip())
            file_size = float(input("File size (MB): ").strip())
            file_format = input("File format: ").strip().lower()
            ebook = EBook(title, author, publisher, page_count, price, file_size, file_format)
            ebooks.append(ebook)
        except ValueError:
            print("Invalid input. Please try again.")

    # Display all eBooks
    print("\nEBooks in the collection:")
    for ebook in ebooks:
        print(ebook)


if __name__ == "__main__":
    main_ebook()
