import pickle

# Part A: Reading the file and validating account numbers
def load_accounts(filename):
    """Load accounts from a text file into a 2D list."""
    accounts = []
    try:
        with open(filename, 'r') as file:
            lines = file.readlines()[1:]  # Skip header line
            for line in lines:
                account, balance = line.strip().split(',')
                accounts.append([account.strip(), float(balance.strip())])
    except FileNotFoundError:
        print(f"Error: The file '{filename}' was not found.")
    except ValueError:
        print("Error: File contains invalid data.")
    return accounts


def validate_account(accounts):
    """Validate account number entered by the user."""
    while True:
        account_number = input("Enter an account number: ").strip()
        for account, balance in accounts:
            if account == account_number:
                print(f"Account {account_number} found. Balance: £{balance:.2f}")
                return
        print("Invalid account number. Please try again.")

# Part B: Pickling and unpickling the account data
def pickle_accounts(accounts, binary_filename):
    """Write account data to a binary file."""
    try:
        with open(binary_filename, 'wb') as binary_file:
            pickle.dump(accounts, binary_file)
        print(f"Accounts successfully written to {binary_filename}.")
    except Exception as e:
        print(f"Error while writing to binary file: {e}")


def unpickle_accounts(binary_filename):
    """Read account data from a binary file."""
    try:
        with open(binary_filename, 'rb') as binary_file:
            accounts = pickle.load(binary_file)
            print(f"Accounts successfully loaded from {binary_filename}.")
            return accounts
    except FileNotFoundError:
        print(f"Error: The file '{binary_filename}' was not found.")
    except Exception as e:
        print(f"Error while reading from binary file: {e}")
        return []

# Main Program
def main():
    text_filename = "bank_accounts.txt"
    binary_filename = "bank.dat"

    # Part A: Load and validate accounts
    accounts = load_accounts(text_filename)
    if accounts:
        validate_account(accounts)

    # Part B: Pickle and unpickle accounts
    pickle_accounts(accounts, binary_filename)
    loaded_accounts = unpickle_accounts(binary_filename)
    print("Loaded Accounts:")
    for account, balance in loaded_accounts:
        print(f"  Account: {account}, Balance: £{balance:.2f}")


if __name__ == "__main__":
    main()
