def main():
    while True:
        try:
            # Prompt the user to enter an even number
            number = int(input("Enter an even number: "))

            # Check if the number is even
            if number % 2 != 0:
                print("The number entered is not even. Please try again.")
                continue

            # Calculate the sum of all even numbers up to the entered value
            total = sum(i for i in range(2, number + 1, 2))

            # Print the result
            print(f"The sum of all even numbers up to {number} is {total}.")
            break
        except ValueError:
            print("Invalid input. Please enter a valid number.")

if __name__ == "__main__":
    main()
