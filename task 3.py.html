def T1(n):
    return 0.002 * n ** 3 + 0.005 * n **2

def T2(n):
    return 0.36 * n + 0.15

# Find the value of n where T2(n) < T1(n)
def find_crossover_point():
    n = 1
    while T2(n) >= T1(n):
        n += 1
    return n

if __name__ == "__main__":
    crossover_n = find_crossover_point()
    print(f"Algorithm A2 becomes more efficient than A1 as = {crossover_n}")
