class EBook(Book):
    """A subclass representing an eBook."""

    def __init__(self, title, author, publisher, page_count, price, file_size, file_format):
        super().__init__(title, author, publisher, page_count, price)
        self.file_size = file_size
        self.file_format = file_format

    def __str__(self):
        return (super().__str__() +
                f", File Size: {self.file_size}MB, Format: {self.file_format}")


# Test program for the EBook class
def main_ebook():
    ebooks = []
    print("Enter details of eBooks (type 'stop' to finish):")
    while True:
        title = input("Title: ").strip()
        if title.lower() == "stop":
            break
        author = input("Author: ").strip()
        publisher = input("Publisher: ").strip()
        try:
            page_count = int(input("Page count: ").strip())
            price = float(input("Price: ").strip())
            file_size = float(input("File size (MB): ").strip())
            file_format = input("File format: ").strip().lower()
            ebook = EBook(title, author, publisher, page_count, price, file_size, file_format)
            ebooks.append(ebook)
        except ValueError:
            print("Invalid input. Please try again.")

    # Display all eBooks
    print("\nEBooks in the collection:")
    for ebook in ebooks:
        print(ebook)


if __name__ == "__main__":
    main_ebook()
