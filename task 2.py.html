def process_optometry_file(filename):
    try:
        with open(filename, 'r') as file:
            # Skip the header line
            lines = file.readlines()[1:]

        # Initialize variables
        total_age = 0
        count = 0
        eye_color_counts = {} # creates an empty dictionary to store data

        # Process each line in the file
        for line in lines:
            line = line.strip()
            if line:  # Ignore empty lines
                eye_color, age = line.split(',')
                age = int(age.strip()) #converts cleaned string into integer
                eye_color = eye_color.strip()

                # Update total age and count
                total_age += age
                count += 1

                # Update eye color count
                if eye_color in eye_color_counts: # checks if colour exists in dictionary
                    eye_color_counts[eye_color] += 1 # if it does exist in increases count by 1
                else:
                    eye_color_counts[eye_color] = 1 # if it doesnt exist add it with an intial count of 1

        # Calculate the average age
        average_age = total_age / count

        # Calculate the percentage for each eye color
        percentages = {
            color: (count / len(lines)) * 100 for color, eye_count in eye_color_counts.items()
        }

        # Output results
        print(f"Average age: {average_age:.1f}")
        print("Eye color percentages:")
        for color, percentage in percentages.items():
            print(f"  {color}: {percentage:.1f}%")

    except:
        print(f"{filename} not found")


    # Replace 'optometry.txt' with the path to your file
process_optometry_file('optometry.txt')
